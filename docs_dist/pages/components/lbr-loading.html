# loading加载

该组件只能全局引用

# 调用方法

// 显示loading
this.$lbrLoading.show('正在加载中呢');   
// 隐藏loading
this.$lbrLoading.hide();

//可根据window进行访问
window.$lbrLoading.show();
window.$lbrLoading.hide();
